[3]:

import base64

def save(file_name, data):
    with open(file_name, "wb") as f:
        data = base64.b64encode(data)
        f.write(data)

def load(file_name):
    with open(file_name, "rb") as f:
        data = f.read()
        data = base64.b64decode(data)
        return data
